SIT.PORTALRM.002 - Retorna Acessos aos Menus
    "SELECT 
    -- Verifica acesso à Solicitação de Compras
    CASE 
        WHEN EXISTS (
        SELECT 1
        FROM TUSRTMV tu
        JOIN TTMV tt
            ON tu.CODCOLIGADA = tt.CODCOLIGADA
        AND tu.CODTMV = tt.CODTMV
        WHERE tu.CODUSUARIO = :CODUSUARIO
            AND tt.CLASSIFICACAO = '04.06.01'
        ) THEN 1
        ELSE 0
    END AS MNUSC,

    -- Verifica acesso ao menu específico
    CASE 
        WHEN EXISTS (
        SELECT 1
        FROM GAUTZMENU a
        LEFT JOIN GUSRPERFIL b
            ON a.CODSISTEMA = b.CODSISTEMA AND a.CODPERFIL = b.CODPERFIL
        LEFT JOIN GUSUARIO c
            ON b.CODUSUARIO = c.CODUSUARIO
        WHERE a.CODSISTEMA = 'F'
            AND a.TAGMENU IN ('810000029','810000041','810001000')
            AND a.PERMISSOES IN ('*','**')
            AND c.CODUSUARIO = :CODUSUARIO
        ) THEN 1
        ELSE 0
    END AS MNULF,

    -- Verifica aceso nos menus do Labore
    CASE 
        WHEN EXISTS (
        SELECT 1
        FROM GAUTZMENU a
        LEFT JOIN GUSRPERFIL b
            ON a.CODSISTEMA = b.CODSISTEMA AND a.CODPERFIL = b.CODPERFIL
        LEFT JOIN GUSUARIO c
            ON b.CODUSUARIO = c.CODUSUARIO
        WHERE a.CODSISTEMA = 'P'
            /*AND a.TAGMENU IN ('810000029','810000041','810001000')
            AND a.PERMISSOES IN ('*','**')*/
            AND c.CODUSUARIO = :CODUSUARIO
        ) THEN 1
        ELSE 0
    END AS MNULB"

SIT.PORTALRM.021 - Retorna Movimentos por Tipo de Movimento
    "SELECT *
    FROM TMOV tm
    left join FCFO f
    on f.codcoligada = tm.codcolcfo and f.codcfo = tm.codcfo
    WHERE tm.CODCOLIGADA = :CODCOLIGADA
    AND ',' + :CODTMV + ',' LIKE '%,' + tm.CODTMV + ',%'"
